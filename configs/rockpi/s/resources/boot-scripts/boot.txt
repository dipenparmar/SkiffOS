# setenv silent 1
setenv condev "console=ttyFIQ0,115200n8"
setenv verify 0

setenv kernel_addr_r "0x40008000"
setenv initramfs_addr_r "0x46000000"
setenv dtb_addr_r "0x44000000"

# Load kernel, initrd and dtb in that sequence
fatload mmc 0:4 ${kernel_addr_r} Image
fatload mmc 0:4 ${initramfs_addr_r} rootfs.cpio.uboot
fatload mmc 0:4 ${dtb_addr_r} rockpi-s-linux.dtb

# final boot args
setenv bootargs "console=tty1 ${condev} root=/dev/initrd ro ramdisk_size=100000 fsck.repair=yes net.ifnames=0 no_console_suspend"

# Boot the board
bootz ${kernel_addr_r} ${initramfs_addr_r} ${dtb_addr_r}
